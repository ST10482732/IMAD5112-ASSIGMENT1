package vcmsa.ci.mealapp

import android.os.Bundle
import android.view.View
import android.widget.Button
import android.widget.EditText
import android.widget.TextView
import android.widget.Toast
import androidx.activity.enableEdgeToEdge
import androidx.appcompat.app.AppCompatActivity
import androidx.core.view.ViewCompat
import androidx.core.view.WindowInsetsCompat


class MainActivity : AppCompatActivity() {

    var timeInput : EditText? = null //(Anon,2025)
    var DisplayButton : Button? = null //(Anon,2025)
    var ResetButton : Button? = null //(Anon,2025)
    var output : TextView? = null //(Anon,2025)


    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        enableEdgeToEdge()
        setContentView(R.layout.activity_main)

        timeInput = findViewById<View>(R.id.timeInput) as EditText //(Anon,2017)
        DisplayButton = findViewById<View>(R.id.displayBtn) as Button //(Anon,2017)
        ResetButton = findViewById<View>(R.id.resetBtn) as Button //(Anon,2017)
        output = findViewById<View>(R.id.output) as TextView //(Anon,2017)

        ResetButton!!.setOnClickListener{ //(Agarwal,2017)
            timeInput!!.text.clear()
        }
        DisplayButton!!.setOnClickListener { //(Agarwal,2017)
           timeInput!!.text.toString()
            DisplayButton!!.text.toString()
        }

        DisplayButton!!.setOnClickListener { //(Agarwal,2017)
            val timeOfDay = timeInput!!.text.toString()
            val mealSuggestion : String =if //(Anon,2025)
             (timeOfDay == "Morning") {
                "Breakfast suggestion: Bread with eggs"
            } else if (timeOfDay == "Mid-morning") {
                "Light snack: Fruit and Nut Butter"
            } else if (timeOfDay == "Afternoon") {
                "lunch suggestion: Sweet and Spicy Turkey Sandwich"
            } else if (timeOfDay == "Mid-afternoon") {
                "Quick bite suggestion: Cinnamon Roll Poke Cake"
            } else if (timeOfDay == "Dinner") {
                "Main course suggestion: Pasta With Marinara Sauce"
            } else if (timeOfDay == "After dinner") {
                "Desert suggestion: Lemon IceCream"
            } else {
                "Invalid input.Please enter a valid time of day"
            }
            output!!.text = mealSuggestion


                ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main)) { v, insets ->
                    val systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars())
                    v.setPadding(
                        systemBars.left,
                        systemBars.top,
                        systemBars.right,
                        systemBars.bottom
                    )
                    insets
                }}}}
